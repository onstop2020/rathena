// THE BOX 5, Bonus Divder 7 Training Dummy Area
//==================================================
-	script	Force2dot2	-1,{
OnMonsterDead:
	if(!playerattached())
	end;
	
	areamonster .mapName$,162,161,185,150,"--ja--",.mobIds[rand(.mobIdSize)],1,strnpcinfo(0) + "::OnMonsterDead";
	setunitdata $@mobid[0],UMOB_DYNAMIC,25;
	setunitdata $@mobid[0],UMOB_DEFLECT,50;
	
	end;
	
	L_Spawn:
	for(.@i = 0; .@i < .spawn_amount; .@i++){
		areamonster .mapName$,162,161,185,150,"--ja--",.mobIds[rand(.mobIdSize)],1,strnpcinfo(0) + "::OnMonsterDead";
		setunitdata $@mobid[0],UMOB_DYNAMIC,25;
		setunitdata $@mobid[0],UMOB_DEFLECT,50;
	}
	
	return;
	
OnInit:
	setarray .mobIds,2409,2410,2411;
	.mobIdSize = getarraysize(.mobIds);
	
	.spawn_amount = 7;
	
	.mapName$ = "force_2-2";
	
	callsub L_Spawn;
	
	end;
}
