-	script	LoginEvent	-1,{
	end;
	
OnPCLoginEvent:
	// New character
	if((BaseLevel == 1)
			&& (JobLevel == 1)
			&& (Class == 0))
	percentheal 100,100;
	
	// Unexpected error
	setpcblock PCBLOCK_ALL,false;
	
	// Important loots
	atcommand "@alootid +10000006";
	
	if(debuff)
	atcommand "@debuff " + debuff;
	
	end;
}
