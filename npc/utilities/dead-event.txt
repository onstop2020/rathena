-	script	DeadEvent	-1,{
	end;
	
OnPCDieEvent:
	Reborn();
	if(!isPassTutorial){
		isPassTutorial = 1;
		savepoint "prontera",156,191,5,5;
		getitem 10000000,1;
		getitem 10000001,1;
	}
	// Back to town when dead
	else
	addtimer 1000,strnpcinfo(3) + "::OnBackToTown";
	end;	
	
OnBackToTown:
	warp "prontera",156,191;
	end;
}
